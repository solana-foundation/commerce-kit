# React SDK

The `@solana-commerce/react-sdk` package provides comprehensive e-commerce React components that make it easy to integrate Solana Pay into your applications. From simple tip buttons to full shopping cart experiences, it includes everything you need.

## Quick Start

### Complete E-commerce Solution

```tsx
import { SolanaCommerceSDK } from '@solana-commerce/react-sdk';

function MyStore() {
  return (
    <SolanaCommerceSDK
      config={{
        mode: 'cart',
        position: 'overlay',
        merchant: {
          name: 'My Store',
          wallet: 'YOUR_SOLANA_ADDRESS',
          description: 'Premium digital products'
        },
        theme: {
          primaryColor: '#9945FF',
          borderRadius: 'lg'
        },
        products: [
          {
            id: 'item-1',
            name: 'Premium Course',
            description: 'Complete blockchain development guide',
            price: 100000000, // 0.1 SOL
            currency: 'SOL'
          }
        ],
        allowedMints: ['SOL', 'USDC'],
        showQR: true,
        showProductDetails: true
      }}
      onPayment={(amount, currency, products) => {
        console.log('Payment initiated:', { amount, currency, products });
      }}
      onPaymentSuccess={(signature) => {
        console.log('Payment successful!', signature);
      }}
    />
  );
}
```

## Component Reference

### `SolanaCommerceSDK`

<FullPageBuyNow2 />

The comprehensive e-commerce component that supports multiple payment modes, theming, and product management.

**Props:**

```typescript
interface SolanaCommerceSDKProps {
  config: SolanaCommerceConfig;
  children?: React.ReactNode; // Custom trigger element
  className?: string;
  style?: React.CSSProperties;
  // Callbacks
  onPayment?: (amount: number, currency: string, products?: Product[]) => void;
  onPaymentStart?: () => void;
  onPaymentSuccess?: (signature: string) => void;
  onPaymentError?: (error: Error) => void;
  onCancel?: () => void;
}

interface SolanaCommerceConfig {
  mode: 'cart' | 'tip' | 'buyNow';
  position?: 'inline' | 'overlay';
  merchant: MerchantConfig;
  theme?: ThemeConfig;
  products?: Product[];
  allowedMints?: string[]; // ['SOL', 'USDC', 'USDT']
  network?: 'mainnet-beta' | 'devnet' | 'testnet';
  // Payment options
  showQR?: boolean;
  enableWalletConnect?: boolean;
  // UI options
  showMerchantInfo?: boolean;
  showProductDetails?: boolean;
  customStyles?: React.CSSProperties;
}

interface MerchantConfig {
  name: string;
  wallet: string; // recipient wallet address
  logo?: string;
  description?: string;
}

interface ThemeConfig {
  primaryColor?: string;
  secondaryColor?: string;
  backgroundColor?: string;
  textColor?: string;
  borderRadius?: 'none' | 'sm' | 'md' | 'lg' | 'xl';
  fontFamily?: string;
}

interface Product {
  id: string;
  name: string;
  description?: string;
  price: number; // in lamports
  currency?: string; // mint address or 'SOL'
  image?: string;
  metadata?: Record<string, any>;
}
```

## Payment Modes

### Tip Mode

Perfect for donations, tips, and simple payments:

```tsx
<SolanaCommerceSDK
  config={{
    mode: 'tip',
    position: 'overlay',
    merchant: {
      name: 'Creator Name',
      wallet: 'YOUR_WALLET_ADDRESS'
    },
    theme: {
      primaryColor: '#9945FF',
      borderRadius: 'lg'
    },
    products: [{
      id: 'tip-1',
      name: 'Support Tip',
      description: 'Thank you for your support!',
      price: 10000000, // 0.01 SOL
      currency: 'SOL'
    }],
    showProductDetails: false
  }}
  onPayment={(amount, currency) => {
    console.log(`Tip: ${amount} ${currency}`);
  }}
/>
```

### Buy Now Mode

Single product purchases with detailed information:

```tsx
<SolanaCommerceSDK
  config={{
    mode: 'buyNow',
    position: 'overlay',
    merchant: {
      name: 'Digital Store',
      wallet: 'YOUR_WALLET_ADDRESS',
      description: 'Premium digital products'
    },
    theme: {
      primaryColor: '#14F195',
      borderRadius: 'md'
    },
    products: [{
      id: 'course-1',
      name: 'Solana Development Course',
      description: 'Complete guide to building on Solana',
      price: 100000000, // 0.1 SOL
      currency: 'SOL'
    }],
    allowedMints: ['SOL', 'USDC'],
    showProductDetails: true
  }}
  onPayment={(amount, currency, products) => {
    console.log('Purchase:', products?.[0]);
  }}
/>
```
