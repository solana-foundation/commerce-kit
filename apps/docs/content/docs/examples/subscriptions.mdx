# Subscription Billing

Build a complete subscription system with recurring payments, plan management, and billing automation. This example shows how to create different subscription tiers and handle payment cycles.

## Overview

We'll implement:
- **Multiple subscription plans** with different features
- **Payment cycle management** (monthly, yearly)
- **Usage tracking** and billing
- **Plan upgrades/downgrades**
- **Payment failure handling**

## Complete Example

import { createCommerceClient } from '@solana-commerce/headless-sdk';

export async function createSubscription(userId: string, planId: string, transactionSignature: string) {
  // 1. Verify the payment transaction
  const client = createCommerceClient({ network: 'mainnet' });
  const transaction = await client.rpc.getTransaction(transactionSignature).send();
  
  if (!transaction) {
    throw new Error('Transaction not found');
  }
  
  // 2. Create subscription record
  const subscription = {
    id: generateId(),
    userId,
    planId,
    status: 'active',
    currentPeriodStart: new Date(),
    currentPeriodEnd: getNextBillingDate(planId),
    transactionSignature,
    createdAt: new Date()
  };
  
  // 3. Save to database
  await saveSubscription(subscription);
  
  // 4. Set up recurring billing
  await scheduleNextPayment(subscription);
  
  return subscription;
}

export async function processRecurringPayment(subscriptionId: string) {
  const subscription = await getSubscription(subscriptionId);
  
  if (!subscription || subscription.status !== 'active') {
    return;
  }
  
  try {
    // Create payment request
    const paymentRequest = createSolanaPayRequest({
      recipient: process.env.MERCHANT_WALLET!,
      amount: subscription.nextPaymentAmount * 1_000_000_000,
      memo: `Subscription renewal: ${subscription.planId}`
    });
    
    // Send payment notification to user
    await sendPaymentNotification(subscription.userId, paymentRequest.url);
    
    // Update subscription period
    await extendSubscriptionPeriod(subscriptionId);
    
  } catch (error) {
    // Handle payment failure
    await handlePaymentFailure(subscriptionId, error);
  }
}
